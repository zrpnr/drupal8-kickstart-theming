 /**
  * Implements hook_theme_suggestions_HOOK_alter().
  */
 function mytheme_theme_suggestions_field_alter(array &$suggestions, array $variables) {
   $element = $variables['element']['#field_name'];

   switch ($element) {
     case 'field_books':
     case 'field_movies':
       $suggestions[] = 'field__ul';
       break;
     case 'field_top_ten' :
       $suggestions[] = 'field__ol';
       break;
   }
 }
all your preprocesses now go in the mytheme.theme file instead of template.php. You can add template suggestions directly here- so that instead of being stuck with the suggestions you just saw in the debug output, you can add your own unique template names for conditions you specifiy in php. Adding a suggestion also lets you use that suggestion in a custom preprocess hook.
